import { json } from "../../../lib/response";
import { requireAuth, requireRole } from "../../../lib/auth";

export async function onRequestPut(context: { request: Request; env: { DB?: D1Database; JWT_SECRET?: string; ENVIRONMENT?: string }; params: { id: string } }): Promise<Response> {
  try {
    const a = await requireAuth(context.request, context.env);
    if (a instanceof Response) return a;
    const rc = requireRole(a, "platform_owner");
    if (rc) return rc;
    const body = await context.request.json() as { address?: string; lat?: number; lng?: number; score?: number };
    const parts: string[] = [];
    const values: (string | number)[] = [];
    if (body.address !== undefined) { parts.push("address = ?"); values.push(body.address); }
    if (typeof body.lat === "number") { parts.push("latitude = ?"); values.push(body.lat); }
    if (typeof body.lng === "number") { parts.push("longitude = ?"); values.push(body.lng); }
    if (typeof body.score === "number") { parts.push("score = ?"); values.push(body.score); }
    if (parts.length === 0) return json({ error: "No fields to update" }, 400);
    values.push(context.params.id);
    await context.env.DB!.prepare(`UPDATE inspiration SET ${parts.join(", ")} WHERE id = ?`).bind(...values).run();
    return json({ success: true });
  } catch (err) {
    console.error("Admin update pin error:", err);
    return json({ error: "Something went wrong" }, 500);
  }
}

export async function onRequestDelete(context: { request: Request; env: { DB?: D1Database; JWT_SECRET?: string; ENVIRONMENT?: string }; params: { id: string } }): Promise<Response> {
  try {
    const a = await requireAuth(context.request, context.env);
    if (a instanceof Response) return a;
    const rc = requireRole(a, "platform_owner");
    if (rc) return rc;
    const db = context.env.DB!;
    await db.prepare("DELETE FROM inspiration_loves WHERE inspiration_id = ?").bind(context.params.id).run();
    await db.prepare("DELETE FROM inspiration WHERE id = ?").bind(context.params.id).run();
    return json({ success: true });
  } catch (err) {
    console.error("Admin delete pin error:", err);
    return json({ error: "Something went wrong" }, 500);
  }
}

--- STEP D: EXTEND functions/api/photographers.ts onRequestPut ---
Current PUT rejects anything that isn't { id, status }. The admin UI also
sends { id, verified }, { id, featured }, { id, slug }, { id, tagline }.
Replace onRequestPut with:

export async function onRequestPut(context: { request: Request; env: { DB?: D1Database; JWT_SECRET?: string; ENVIRONMENT?: string; DEEPSEEK_API_KEY?: string } }): Promise<Response> {
  try {
    const a = await requireAuth(context.request, context.env);
    if (a instanceof Response) return a;
    const rc = requireRole(a, "platform_owner");
    if (rc) return rc;
    const body = await context.request.json() as { id: string; status?: string; verified?: boolean; featured?: boolean; slug?: string; tagline?: string };
    if (!body.id) return json({ error: "id is required" }, 400);
    const db = context.env.DB!;
    const parts: string[] = [];
    const values: (string | number)[] = [];
    if (body.status !== undefined) {
      if (!["pending", "approved", "rejected"].includes(body.status)) return json({ error: "Invalid status" }, 400);
      parts.push("status = ?"); values.push(body.status);
    }
    if (body.verified !== undefined) {
      parts.push("verified = ?"); values.push(body.verified ? 1 : 0);
      parts.push("verified_at = ?"); values.push(body.verified ? new Date().toISOString() : "");
    }
    if (body.featured !== undefined) { parts.push("featured = ?"); values.push(body.featured ? 1 : 0); }
    if (body.slug !== undefined) {
      if (!/^[a-z0-9]+(?:-[a-z0-9]+)*$/.test(body.slug)) return json({ error: "Invalid slug format" }, 400);
      const existing = await db.prepare("SELECT id FROM photographers WHERE slug = ? AND id != ?").bind(body.slug, body.id).first();
      if (existing) return json({ error: "Slug already taken" }, 409);
      parts.push("slug = ?"); values.push(body.slug);
    }
    if (body.tagline !== undefined) {
      if (body.tagline.length > 200) return json({ error: "Tagline too long" }, 400);
      parts.push("tagline = ?"); values.push(body.tagline);
    }
    if (parts.length === 0) return json({ error: "No fields to update" }, 400);
    values.push(body.id);
    await db.prepare(`UPDATE photographers SET ${parts.join(", ")} WHERE id = ?`).bind(...values).run();
    return json({ success: true });
  } catch (err) {
    console.error("Update photographer error:", err);
    return json({ error: "Something went wrong." }, 500);
  }
}

async function handleResponse<T>(res: Response): Promise<T> {
  const ct = res.headers.get("content-type") || "";
  if (ct.includes("text/html")) {
    // An /api call answered by the SPA shell = endpoint does not exist.
    throw new Error(`API route not found (${res.status}). Check the endpoint path.`);
  }
  if (!ct.includes("application/json")) {
    if (!res.ok) throw new Error(`Request failed (${res.status})`);
    return {} as T; // 204 No Content etc.
  }
  const data = await res.json();
  if (!res.ok) throw new Error((data as ApiError).error ?? `Request failed (${res.status})`);
  return data as T;
}

export async function verifyMediaSignature(storageKey: string, signature: string, expires: string, secret: string): Promise<boolean> {
  const exp = parseInt(expires, 10);
  if (isNaN(exp) || exp < Math.floor(Date.now() / 1000)) return false; // expired
  if (!/^[0-9a-f]+$/i.test(signature) || signature.length !== 64) return false;
  const data = `${storageKey}:${exp}`;
  const key = await crypto.subtle.importKey("raw", new TextEncoder().encode(secret), { name: "HMAC", hash: "SHA-256" }, false, ["verify"]);
  const sigBytes = new Uint8Array(signature.match(/.{2}/g)!.map((b) => parseInt(b, 16)));
  return crypto.subtle.verify("HMAC", key, sigBytes, new TextEncoder().encode(data));
}
